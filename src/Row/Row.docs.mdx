# Row
A horizontal stack. For a vertical stack, see \<Stack />.

<Row>
  <button>Save</button>
  <button>Delete</button>
  <button>Share</button>
</Row>

Row composes nicely with Split to create split layouts:

<Example1 />

export const Example1 = () => {
  return (
    <Row gap={1.5}>
      <b>
        Awesome toolbar
      </b>
      <button>
        button 1
      </button>
      <button>
        button 2
      </button>
      <Split />
      <button>A button</button>
    </Row>
  )
}
