Row composes nicely with Split to create split layouts:

<Example1 />

export const Example1 = () => {
  return (
    <Row gap={1.5}>
      <b>
        Awesome toolbar
      </b>
      <button>
        button 1
      </button>
      <button>
        button 2
      </button>
      <Split />
      <button>A button</button>
    </Row>
  )
}

With the `wrap` prop you can also achieve common "pill" style inline metadata/tags/etc:

<Example2 />

export const Example2 = () => {
  return (
    <Clamp clamp='380px'>
      <Row wrap>
        <Pill>Published 01.01.1970</Pill>
        <Pill>Christian Kaindl</Pill>
        <Pill>JavaScript</Pill>
        <Pill>CSS</Pill>
        <Pill>React</Pill>
        <Pill>package</Pill>
        <Pill>layout</Pill>
        <button>Share</button>
      </Row>
    </Clamp>
  )
}

export const Pill = ({ children }) => {
  return (
    <span style={{ borderRadius: 99, backgroundColor: 'rgb(0, 0, 0, .08)', padding: '4px 9px' }}>
      {children}
    </span>
  )
}

The `expandChildren` prop let's you customize the columns `flex-grow` values:

<Example3 />

export const Example3 = () => {
  return (
    <Row gap={1.5} expandChildren>
      <Box>Column 1</Box>
      <Box>Column 2</Box>
      <Box>Column 3</Box>
      <Box>Column 4</Box>
    </Row>
  )
}

Or with custom ratios:

<Example4 />

export const Example4 = () => {
  return (
    <Row gap={1.5} expandChildren='1/3/0/0'>
      <Box>Column 1</Box>
      <Box>Column 2</Box>
      <Box>Column 3</Box>
      <Box>Column 4</Box>
    </Row>
  )
}
