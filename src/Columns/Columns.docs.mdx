# Columns
Columns its children to a max-width and centers them.

export const Example = () => {
  return (
    <Columns>
      <Stack bleed='10px 5px' bleedLeft='7px'>
        <h3>Heading</h3>
        <p>Lorem ipsum</p>
      </Stack>
      <Stack>
        <h3>Heading</h3>
        <p>Lorem ipsum</p>
      </Stack>
    </Columns>
  )
}
